问题 1.1

$$
\begin{align*}
&\lim_{x \to +\infty} \frac{\sqrt[3]{x^2 \sqrt{x^2 + 1}} + 3x}{x}
\\=& \lim_{x \to +\infty} \frac{\sqrt[3]{x^2 \cdot \sqrt{x^2(1 + \frac{1}{x^2})}} + 3x}{x}
\\=& \lim_{x \to +\infty} \frac{\sqrt[3]{x^2 \cdot x \sqrt{1 + \frac{1}{x^2}}} + 3x}{x}
\\=& \lim_{x \to +\infty} \frac{\sqrt[3]{x^3 \sqrt{1 + \frac{1}{x^2}}} + 3x}{x}
\\=& \lim_{x \to +\infty} \left( \frac{\sqrt[3]{x^3 \sqrt{1 + \frac{1}{x^2}}}}{x} + \frac{3x}{x} \right)
\\=& \lim_{x \to +\infty} \left( \sqrt[3]{\frac{x^3 \sqrt{1 + \frac{1}{x^2}}}{x^3}} + 3 \right)
\\=& \lim_{x \to +\infty} \left( \sqrt[3]{\sqrt{1 + \frac{1}{x^2}}} + 3 \right)(当 x \to +\infty 时：\Rightarrow \frac{1}{x^2} \to 0,)
\\=& \sqrt[3]{\sqrt{1 + 0}} + 3
\\=& \sqrt[3]{1} + 3
\\=& 1 + 3
\\=& 4.
\end{align*}
$$

问题 1.2

$$
\begin{align*}
&\lim_{x \to +\infty} \left( \sqrt{x^2 + 3x + 1} - \sqrt{x^2 - 3x + 1} \right)
\\=&\lim_{x \to +\infty} \frac{\left( \sqrt{x^2 + 3x + 1} - \sqrt{x^2 - 3x + 1} \right) \cdot \left( \sqrt{x^2 + 3x + 1} + \sqrt{x^2 - 3x + 1} \right)}{\sqrt{x^2 + 3x + 1} + \sqrt{x^2 - 3x + 1}}
\\=&\lim_{x \to +\infty} \frac{\left( x^2 + 3x + 1 - (x^2 - 3x + 1) \right)}{\sqrt{x^2 + 3x + 1} + \sqrt{x^2 - 3x + 1}}
\\=&\lim_{x \to +\infty} \frac{6x}{\sqrt{x^2 + 3x + 1} + \sqrt{x^2 - 3x + 1}}
\\=&\lim_{x \to +\infty} \frac{6x}{\sqrt{x^2(1 + \frac{3}{x} + \frac{1}{x^2})} + \sqrt{x^2(1 - \frac{3}{x} + \frac{1}{x^2})}}
\\=&\lim_{x \to +\infty} \frac{6x}{x\sqrt{1 + \frac{3}{x} + \frac{1}{x^2}} + x\sqrt{1 - \frac{3}{x} + \frac{1}{x^2}}}
\\=&\lim_{x \to +\infty} \frac{6}{\sqrt{1 + \frac{3}{x} + \frac{1}{x^2}} + \sqrt{1 - \frac{3}{x} + \frac{1}{x^2}}}
\\=&\frac{6}{\sqrt{1 + 0} + \sqrt{1 + 0}}
\\=&\frac{6}{2} = 3.
\end{align*}
$$

问题 1.3

$$
\begin{align*}
&\lim_{x \to 0} \frac{x^4 - \sin x + \tan x}{\ln(1 + x^3)}
\\=&\lim_{x \to 0} \frac{x^4 - \left(x - \frac{x^3}{6} + o(x^3)\right) + \left(x + \frac{x^3}{3} + o(x^3)\right)}{x^3 + o(x^3)}
\\=&\lim_{x \to 0} \frac{x^4 - x + \frac{x^3}{6} + x + \frac{x^3}{3} + o(x^3)}{x^3 + o(x^3)}
\\=&\lim_{x \to 0} \frac{x^4 + \frac{x^3}{6} + \frac{x^3}{3} + o(x^3)}{x^3 + o(x^3)}
\\=&\lim_{x \to 0} \frac{x^4 + \frac{x^3}{2} + o(x^3)}{x^3 + o(x^3)}
\\=&\lim_{x \to 0} \left( \frac{x^4}{x^3} + \frac{\frac{x^3}{2}}{x^3} + \frac{o(x^3)}{x^3} \right)
\\=&\lim_{x \to 0} \left( x + \frac{1}{2} + o(1) \right)
\\=&\frac{1}{2}.
\end{align*}
$$

问题 1.5

$$
\begin{align*}
&\lim_{x \to +\infty} \frac{(2x^3 + 7x - 1)^6}{(2x^6 - 13x^2 + x)^3}
\\=&\lim_{x \to +\infty} \frac{\left(x^3 \cdot \left(2 + \frac{7}{x^2} - \frac{1}{x^3}\right)\right)^6}{\left(x^6 \cdot \left(2 - \frac{13}{x^4} + \frac{1}{x^5}\right)\right)^3}
\\=&\lim_{x \to +\infty} \frac{\left(x^3\right)^6 \cdot \left(2 + \frac{7}{x^2} - \frac{1}{x^3}\right)^6}{\left(x^6\right)^3 \cdot \left(2 - \frac{13}{x^4} + \frac{1}{x^5}\right)^3}
\\=&\lim_{x \to +\infty} \frac{x^{18} \cdot \left(2 + \frac{7}{x^2} - \frac{1}{x^3}\right)^6}{x^{18} \cdot \left(2 - \frac{13}{x^4} + \frac{1}{x^5}\right)^3}
\\=&\lim_{x \to +\infty} \frac{\left(2 + \frac{7}{x^2} - \frac{1}{x^3}\right)^6}{\left(2 - \frac{13}{x^4} + \frac{1}{x^5}\right)^3}
\\=&\frac{\left(2 + 0 - 0\right)^6}{\left(2 - 0 + 0\right)^3}

\end{align*}

问题 1.6

$$
\begin{align*}
&\lim_{x \to 0} \frac{x}{\sqrt[3]{1 + x} - 1}
\\=&\lim_{x \to 0} \frac{x \cdot \left(\sqrt[3]{(1 + x)^2} + \sqrt[3]{1 + x} + 1\right)}{\left(\sqrt[3]{1 + x} - 1\right) \cdot \left(\sqrt[3]{(1 + x)^2} + \sqrt[3]{1 + x} + 1\right)}
\\=&\lim_{x \to 0} \frac{x \cdot \left(\sqrt[3]{(1 + x)^2} + \sqrt[3]{1 + x} + 1\right)}{(1 + x) - 1}
\\=&\lim_{x \to 0} \frac{x \cdot \left(1 + 1 + 1\right)}{x}
\\=&\lim_{x \to 0} \frac{3x}{x}
\\=&3.
\end{align*}
$$

问题 1.7

$$
\begin{align*}
&\lim_{x \to +\infty} \left( \sqrt{x^2 + 7x + 6} - \sqrt{x^2 + 6x + 7} \right)
\\=&\lim_{x \to +\infty} \frac{\left( \sqrt{x^2 + 7x + 6} - \sqrt{x^2 + 6x + 7} \right) \cdot \left( \sqrt{x^2 + 7x + 6} + \sqrt{x^2 + 6x + 7} \right)}{\sqrt{x^2 + 7x + 6} + \sqrt{x^2 + 6x + 7}}
\\=&\lim_{x \to +\infty} \frac{\left( x^2 + 7x + 6 - (x^2 + 6x + 7) \right)}{\sqrt{x^2 + 7x + 6} + \sqrt{x^2 + 6x + 7}}
\\=&\lim_{x \to +\infty} \frac{x}{\sqrt{x^2 + 7x + 6} + \sqrt{x^2 + 6x + 7}}
\\=&\lim_{x \to +\infty} \frac{x-1}{\sqrt{x^2(1 + \frac{7}{x} + \frac{6}{x^2})} + \sqrt{x^2(1 + \frac{6}{x} + \frac{7}{x^2})}}
\\=&\lim_{x \to +\infty} \frac{x(1- \frac{1}{x})}{x\sqrt{1 + \frac{7}{x} + \frac{6}{x^2}} + x\sqrt{1 + \frac{6}{x} + \frac{7}{x^2}}}
\\=&\lim_{x \to +\infty} \frac{1- \frac{1}{x}}{\sqrt{1 + \frac{7}{x} + \frac{6}{x^2}} + \sqrt{1 + \frac{6}{x} + \frac{7}{x^2}}}
\\=&\frac{1}{\sqrt{1 + 0} + \sqrt{1 + 0}}
\\=&\frac{1}{2}.
\end{align*}
$$

\\=&\lim_{x \to 0} e^{\frac{-3 \cdot 5x}{2x} \cdot \frac{1}{1 - 3 \cdot 0}}
\\=&\lim_{x \to 0} e^{\frac{-15}{2}}
\\=&e^{-\frac{15}{2}}.
\end{align*}
$$

问题 1.29

$$
\begin{align*}
&\lim_{x \to 4} \frac{3 - \sqrt{5 + x}}{1 - \sqrt{5 - x}}
\\=&\lim_{x\to{4}}{\dfrac{\left(\sqrt{5-x}+1\right)\left(3-\sqrt{x+5}\right)\left(\sqrt{x+5}+3\right)}{\left(1-\sqrt{5-x}\right)\left(\sqrt{5-x}+1\right)\left(\sqrt{x+5}+3\right)}}
\\=&\lim_{x\to{4}}{\dfrac{\cancel{\left(x-4\right)}\left(-\sqrt{5-x}-1\right)}{\cancel{\left(x-4\right)}\left(\sqrt{x+5}+3\right)}}
\\=&\lim_{x\to{4}}{\dfrac{-\sqrt{5-x}-1}{\sqrt{x+5}+3}}
\\=&\lim_{x\to{4}}{\dfrac{-\sqrt{5-4}-1}{\sqrt{4+5}+3}}\\=&-\dfrac{1}{3}.
\end{align*}

问题 1.30 *

$$
\begin{align*}
&\lim_{x \to +\infty} \left( \frac{3x^2}{2x + 1} - \frac{6x^2 + x + 2}{4x} \right)
\\=&\lim_{x \to +\infty} \left( \frac{3x^2}{2x + 1} - \frac{6x^2}{4x} - \frac{x}{4x} - \frac{2}{4x} \right)
\\=&\lim_{x \to +\infty} \left( \frac{3x^2}{2x + 1} - \frac{3x^2}{2x} - \frac{1}{4} - \frac{1}{2x} \right)
\\=&\lim_{x \to +\infty} \left( \frac{3x^2 (1 - \frac{1}{2x})}{2x + 1} - \frac{1}{4} \right)
\\=&\lim_{x \to +\infty} \left( \frac{3x^2}{2x} - \frac{1}{4} \right)
\\=&\lim_{x \to +\infty} 1 \rightarrow 1.
\end{align*}

问题 1.31

$$
\begin{align*}
&\lim_{x \to \frac{\pi}{2}} \frac{\sin 2x}{\pi^2 - 4x^2}
\\=&\lim_{x \to \frac{\pi}{2}} \frac{\sin 2x}{(\pi - 2x)(\pi + 2x)}
\\=&\dfrac{1}{2\pi}\lim_{x\to{\frac{\pi}{2}}}{-\dfrac{\sin\left(2x\right)}{2x-\pi}},(\begin{gathered}x=t+\dfrac{\pi}{2}&t=x-\dfrac{\pi}{2}\end{gathered},x\to{\frac{\pi}{2}}\Rightarrow{t\to 0})
\\=&-\dfrac{1}{2\pi}\lim_{t\to{0}}{-\dfrac{\sin\left(2t\right)}{2t}}
\\=&\dfrac{1}{2\pi}\lim_{t\to{0}}{\dfrac{\sin\left(2t\right)}{2t}}\underset{2t\to{0}}{=}\dfrac{1}{2\pi}
\\=&\frac{1}{2\pi}.
\end{align*}

问题 1.32

$$
\begin{align*}
&\lim_{x \to 5} \frac{\cos(\pi x) + 1}{(5 - x) \cos\frac{\pi x}{2}}
\\=&\lim_{x \to 5} \frac{\cos(\pi x) + 1}{(5 - x) \cdot \cos\left(\frac{\pi x}{2}\right)}
\\=&\lim_{x \to 5} \frac{\cos(\pi x) + 1}{(5 - x) \cdot \cos\left(\frac{\pi}{2}(5 - x)\right)} \quad (\text{利用 }\cos\left(\frac{\pi}{2} \cdot x\right) = \cos\left(\frac{\pi}{2}(5 - x)\right))
\\=&\lim_{x \to 5} \frac{\cos(\pi x) + 1}{(5 - x) \cdot (-(5 - x))} \quad (\cos\left(\frac{\pi}{2}(5 - x)\right) \sim -(5 - x))
\\=&\lim_{x \to 5} \frac{-1 + 1}{(5 - x)^2}
\\=&\lim_{x \to 5} \frac{0}{(5 - x)^2}
\\=&0.
\end{align*}

问题 1.33

$$
\begin{align*}
&\lim_{x \to 0^+} \left( 1 + \sqrt{x} \right)^{\frac{\ln x}{\sqrt{x}}}
\\=&\lim_{x \to 0^+} e^{\frac{\ln x}{\sqrt{x}} \cdot \ln(1 + \sqrt{x})}
\\=&\lim_{x \to 0^+} e^{\frac{\ln x \cdot \sqrt{x}}{\sqrt{x} \cdot (1 + \sqrt{x})}} \quad (\text{因为 } \ln(1 + \sqrt{x}) \sim \sqrt{x} \text{ 当 } x \to 0^+)
\\=&\lim_{x \to 0^+} e^{\ln x \cdot \frac{\sqrt{x}}{\sqrt{x} \cdot (1 + \sqrt{x})}}
\\=&\lim_{x \to 0^+} e^{\ln x \cdot \frac{1}{1 + \sqrt{x}}}
\\=&\lim_{x \to 0^+} e^{\ln x} \quad (\text{因为 } \frac{1}{1 + \sqrt{x}} \to 1 \text{ 当 } x \to 0^+)
\\=&\lim_{x \to 0^+} x
\\=&0.
\end{align*}

问题 1.34

$$
\begin{align*}
&\lim_{x \to 0} \frac{\sqrt[3]{x^3 + 1} - \sqrt{x^2 + 1}}{x^2}
\\=&\lim_{x \to 0} \frac{\left(\sqrt[3]{x^3 + 1} - 1\right) - \left(\sqrt{x^2 + 1} - 1\right)}{x^2}
\\=&\lim_{x \to 0} \frac{\frac{x^3}{3} - \frac{x^2}{2}}{x^2} \quad (\text{泰勒展开近似：} \sqrt[3]{x^3 + 1} \sim 1 + \frac{x^3}{3}, \sqrt{x^2 + 1} \sim 1 + \frac{x^2}{2})
\\=&\lim_{x \to 0} \left(\frac{x}{3} - \frac{1}{2}\right)
\\=&\frac{0}{3} - \frac{1}{2}
\\=&-\frac{1}{2}.
\end{align*}
$$

问题 1.35

$$
\begin{align*}
&\lim_{x \to \infty} \left( \sqrt{4x^4 + 13x^2 - 7} - 2x^2 \right)
\\=&\lim_{x \to \infty} \frac{\left( \sqrt{4x^4 + 13x^2 - 7} - 2x^2 \right) \cdot \left( \sqrt{4x^4 + 13x^2 - 7} + 2x^2 \right)}{\sqrt{4x^4 + 13x^2 - 7} + 2x^2}
\\=&\lim_{x \to \infty} \frac{\left( (4x^4 + 13x^2 - 7) - (2x^2)^2 \right)}{\sqrt{4x^4 + 13x^2 - 7} + 2x^2}
\\=&\lim_{x \to \infty} \frac{\left( 4x^4 + 13x^2 - 7 - 4x^4 \right)}{\sqrt{4x^4 + 13x^2 - 7} + 2x^2}
\\=&\lim_{x \to \infty} \frac{13x^2 - 7}{\sqrt{4x^4 + 13x^2 - 7} + 2x^2}
\\=&\lim_{x \to \infty} \frac{x^2 \left( 13 - \frac{7}{x^2} \right)}{x^2 \left( 2 + \sqrt{4 + \frac{13}{x^2} - \frac{7}{x^4}} \right)}
\\=&\lim_{x \to \infty} \frac{13 - \frac{7}{x^2}}{2 + \sqrt{4 + \frac{13}{x^2} - \frac{7}{x^4}}}
\\=&\frac{13}{2 + \sqrt{4}}
\\=&\frac{13}{4}.
\end{align*}
$$

\\=&\lim_{x \to 1^-} \frac{\ln(1 - x)}{1 + 3 \ln\cos\left(\frac{\pi}{2} - \frac{\pi}{2}\right)} \quad (\text{当 } x \to 1^-，\cos\frac{\pi x}{2} \to 0)
\\=&\lim_{x \to 1^-} \frac{\ln(1 - x)}{1 + 3 \ln 0} \quad (\ln 0 \to -\infty)

\end{align*}

\\=&\lim_{x \to 0} \frac{\frac{x}{3} + xe^x + x^6}{5x} \quad (\text{近似 } \ln(1 + u) \sim u \text{ 当 } u \to 0)
\\=&\lim_{x \to 0} \frac{\frac{x}{3} + x(1 + x) + x^6}{5x}
\\=&\lim_{x \to 0} \frac{\frac{x}{3} + x + x^2 + x^6}{5x}

问题 1.38

\begin{align*}
&\lim_{x \to 0} \frac{\arcsin x^4 + 3\ln(1 + e^x \sin x)}{x + \cos x - 1}
\\=&\lim_{x \to 0} \frac{x^4 + 3(e^x \sin x)}{x + \cos x - 1} \quad (\text{近似 } \arcsin x^4 \sim x^4, \ln(1 + u) \sim u \text{ 当 } u \to 0)
\\=&\lim_{x \to 0} \frac{x^4 + 3(e^x \cdot x)}{x - (1 - \cos x)} \quad (\text{使用 } \cos x \sim 1 - \frac{x^2}{2})
\\=&\lim_{x \to 0} \frac{x^4 + 3x}{x - \left(-\frac{x^2}{2}\right)}
\\=&\lim_{x \to 0} \frac{x^4 + 3x}{x + \frac{x^2}{2}}
\\=&\lim_{x \to 0} \frac{x(x^3 + 3)}{x\left(1 + \frac{x}{2}\right)}
\\=&\lim_{x \to 0} \frac{x^3 + 3}{1 + \frac{x}{2}}
\\=&\frac{3}{1 + 0}
\\=&3.
\end{align*}

\\=&\lim_{x \to \infty} \left[(x + 2)\ln\left(1 + \frac{7}{2x - 4}\right)\right] \quad (\text{将分母提取为主项})
\\=&\lim_{x \to \infty} \left[(x + 2) \cdot \frac{7}{2x - 4}\right] \quad (\text{使用近似 } \ln(1 + u) \sim u \text{ 当 } u \to 0)

\\=&\lim_{x \to \infty} \frac{7x + 14}{2x - 4}

问题 1.43

$$
\begin{align*}
&\lim_{x \to \frac{\pi}{4}} \left( \tan x \right)^{\frac{1}{\cos\left(\frac{3\pi}{4} - x\right)}}
\\=&\lim_{x \to \frac{\pi}{4}} e^{\frac{\ln(\tan x)}{\cos\left(\frac{3\pi}{4} - x\right)}} \quad (\text{将幂函数转化为指数形式})
\\=&\lim_{x \to \frac{\pi}{4}} e^{\frac{\ln(\tan x)}{-\sin\left(x - \frac{\pi}{4}\right)}} \quad (\cos\left(\frac{3\pi}{4} - x\right) = -\sin\left(x - \frac{\pi}{4}\right))
\\=&\lim_{x \to \frac{\pi}{4}} e^{\frac{\ln(\tan x)}{-(x - \frac{\pi}{4})}} \quad (\sin\left(x - \frac{\pi}{4}\right) \sim x - \frac{\pi}{4})
\\=&\lim_{x \to \frac{\pi}{4}} e^{\frac{\ln\left(1 + 2(x - \frac{\pi}{4})\right)}{-(x - \frac{\pi}{4})}} \quad (\tan x \sim 1 + 2(x - \frac{\pi}{4}))
\\=&\lim_{x \to \frac{\pi}{4}} e^{\frac{2(x - \frac{\pi}{4})}{-(x - \frac{\pi}{4})}} \quad (\ln(1 + u) \sim u \text{ 当 } u \to 0)
\\=&\lim_{x \to \frac{\pi}{4}} e^{-2}
\\=&e^{-2}.
\end{align*}

问题 1.45

$$
\begin{align*}
&\lim_{x \to \infty} \left(x + 3^x\right)^{\frac{2}{x + 3}}
\\=&\lim_{x \to \infty} e^{\frac{2}{x + 3} \ln(x + 3^x)} \quad (\text{将幂函数转化为指数形式})
\\=&\lim_{x \to \infty} e^{\frac{2}{x + 3} \ln\left(3^x\left(1 + \frac{x}{3^x}\right)\right)} \quad (\text{提取 } 3^x \text{ 为主项})
\\=&\lim_{x \to \infty} e^{\frac{2}{x + 3} \left(x \ln 3 + \ln\left(1 + \frac{x}{3^x}\right)\right)} \quad (\ln(ab) = \ln a + \ln b)
\\=&\lim_{x \to \infty} e^{\frac{2}{x + 3} \cdot x \ln 3} \quad (\text{因为 } \ln\left(1 + \frac{x}{3^x}\right) \to 0 \text{ 当 } x \to \infty)
\\=&\lim_{x \to \infty} e^{\frac{2x \ln 3}{x + 3}}
\\=&\lim_{x \to \infty} e^{2 \ln 3 \cdot \frac{x}{x + 3}} \quad (\text{分子分母同除 } x)
\\=&\lim_{x \to \infty} e^{2 \ln 3 \cdot 1} \quad (\frac{x}{x + 3} \to 1 \text{ 当 } x \to \infty)
\\=&e^{2 \ln 3}
\\=&3^2
\\=&9.
\end{align*}
